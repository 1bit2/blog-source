## 三、系统调用入口函数

### 3.1 sys_fork - 创建进程

```c
// kernel/fork.c:3215

SYSCALL_DEFINE0(fork)
{
#ifdef CONFIG_MMU
    /*
     * fork()系统调用的内核实现
     * 
     * 特点：
     *   1. 子进程获得父进程地址空间的完整副本(COW)
     *   2. 子进程获得父进程文件描述符表的副本
     *   3. 子进程退出时向父进程发送SIGCHLD信号
     */
    struct kernel_clone_args args = {
        .exit_signal = SIGCHLD,  // 子进程退出信号
        // 其他标志默认为0，表示复制所有资源
    };

    return kernel_clone(&args);
#else
    /* 
     * 无MMU系统不支持fork
     * 因为COW(写时复制)依赖MMU的页表权限机制
     */
    return -EINVAL;
#endif
}
```

### 3.2 sys_vfork - 创建共享内存的子进程

```c
// kernel/fork.c:3242

SYSCALL_DEFINE0(vfork)
{
    /*
     * vfork()系统调用
     * 
     * 与fork()的区别：
     *   1. CLONE_VM: 父子进程共享地址空间(不复制页表)
     *   2. CLONE_VFORK: 父进程阻塞，直到子进程exit()或exec()
     * 
     * 典型用途：
     *   if (vfork() == 0) {
     *       execve("/bin/ls", ...);  // 子进程立即exec
     *       _exit(1);                // exec失败时退出
     *   }
     *   // 父进程在这里恢复执行
     * 
     * 优势：避免fork+exec场景下不必要的页表复制
     * 风险：子进程修改内存会影响父进程！
     */
    struct kernel_clone_args args = {
        .flags      = CLONE_VFORK | CLONE_VM,
        .exit_signal    = SIGCHLD,
    };

    return kernel_clone(&args);
}
```

### 3.3 sys_clone - 通用进程/线程创建

```c
// kernel/fork.c:3287 (以标准参数顺序为例)

SYSCALL_DEFINE5(clone, unsigned long, clone_flags, unsigned long, newsp,
         int __user *, parent_tidptr,
         int __user *, child_tidptr,
         unsigned long, tls)
{
    /*
     * clone()系统调用 - 最灵活的进程/线程创建接口
     * 
     * 参数说明：
     *   clone_flags: 控制共享哪些资源的标志位组合
     *   newsp:       子进程的用户栈指针(创建线程时必须提供)
     *   parent_tidptr: CLONE_PARENT_SETTID时，将子TID写入此地址
     *   child_tidptr:  CLONE_CHILD_SETTID时，将子TID写入此地址
     *   tls:         CLONE_SETTLS时，设置子进程的TLS
     * 
     * 常用标志组合示例：
     * 
     * 1. 创建POSIX线程 (pthread_create):
     *    CLONE_VM | CLONE_FS | CLONE_FILES | CLONE_SIGHAND | 
     *    CLONE_THREAD | CLONE_SYSVSEM | CLONE_SETTLS |
     *    CLONE_PARENT_SETTID | CLONE_CHILD_CLEARTID
     * 
     * 2. 创建容器:
     *    CLONE_NEWPID | CLONE_NEWNS | CLONE_NEWNET | CLONE_NEWUTS |
     *    CLONE_NEWIPC | CLONE_NEWUSER
     */
    struct kernel_clone_args args = {
        .flags      = (lower_32_bits(clone_flags) & ~CSIGNAL),
        .pidfd      = parent_tidptr,
        .child_tid  = child_tidptr,
        .parent_tid = parent_tidptr,
        .exit_signal    = (lower_32_bits(clone_flags) & CSIGNAL),
        .stack      = newsp,
        .tls        = tls,
    };

    return kernel_clone(&args);
}
```

## 四、核心函数详解

### 4.1 kernel_clone - 统一入口

```c
// kernel/fork.c:3074

/*
 * kernel_clone() - 所有进程/线程创建的统一内核入口
 * 
 * 调用关系：
 *   sys_fork()  ─┐
 *   sys_vfork() ─┼──→ kernel_clone() ──→ copy_process()
 *   sys_clone() ─┤                           │
 *   sys_clone3()┘                            ↓
 *                                    wake_up_new_task()
 * 
 * 返回值：
 *   父进程返回: 子进程的PID
 *   子进程返回: 0 (通过copy_thread设置)
 */
pid_t kernel_clone(struct kernel_clone_args *args)
{
    u64 clone_flags = args->flags;
    struct completion vfork;      // vfork完成信号量
    struct pid *pid;
    struct task_struct *p;
    int trace = 0;
    pid_t nr;

    /* 
     * 步骤1: 参数校验
     * CLONE_PIDFD和CLONE_PARENT_SETTID不能共用同一地址
     */
    if ((args->flags & CLONE_PIDFD) &&
        (args->flags & CLONE_PARENT_SETTID) &&
        (args->pidfd == args->parent_tid))
        return -EINVAL;

    /*
     * 步骤2: 确定ptrace事件类型
     * 用于通知调试器(gdb等)进程创建事件
     */
    if (!(clone_flags & CLONE_UNTRACED)) {
        if (clone_flags & CLONE_VFORK)
            trace = PTRACE_EVENT_VFORK;
        else if (args->exit_signal != SIGCHLD)
            trace = PTRACE_EVENT_CLONE;  // 创建线程
        else
            trace = PTRACE_EVENT_FORK;   // 创建进程

        if (likely(!ptrace_event_enabled(current, trace)))
            trace = 0;
    }

    /*
     * 步骤3: 调用copy_process()创建新任务
     * 这是真正的"复制"操作
     */
    p = copy_process(NULL, trace, NUMA_NO_NODE, args);
    add_latent_entropy();

    if (IS_ERR(p))
        return PTR_ERR(p);

    /*
     * 步骤4: 获取PID（必须在唤醒子进程前完成）
     * 子进程可能很快退出，导致task_struct被释放
     */
    trace_sched_process_fork(current, p);
    pid = get_task_pid(p, PIDTYPE_PID);
    nr = pid_vnr(pid);  // 获取当前命名空间中的PID号

    /* CLONE_PARENT_SETTID: 将子PID写回用户空间 */
    if (clone_flags & CLONE_PARENT_SETTID)
        put_user(nr, args->parent_tid);

    /*
     * 步骤5: vfork特殊处理
     * 初始化completion，父进程稍后会在此阻塞
     */
    if (clone_flags & CLONE_VFORK) {
        p->vfork_done = &vfork;
        init_completion(&vfork);
        get_task_struct(p);
    }

    /*
     * 步骤6: 唤醒新进程
     * 将其加入调度器运行队列，从此刻起可被调度执行
     */
    wake_up_new_task(p);

    /* 通知调试器 */
    if (unlikely(trace))
        ptrace_event_pid(trace, pid);

    /*
     * 步骤7: vfork父进程阻塞等待
     * 直到子进程调用exec()或exit()
     */
    if (clone_flags & CLONE_VFORK) {
        if (!wait_for_vfork_done(p, &vfork))
            ptrace_event_pid(PTRACE_EVENT_VFORK_DONE, pid);
    }

    put_pid(pid);
    return nr;  // 返回子进程PID给父进程
}
```

### 4.2 copy_process - 核心复制函数

```c
// kernel/fork.c:2244

/*
 * copy_process() - 进程/线程创建的核心函数
 * 
 * 功能：
 *   1. 分配并初始化新的task_struct
 *   2. 根据clone_flags复制或共享各种资源
 *   3. 将新任务加入进程树和各种哈希表
 * 
 * 资源复制/共享决策表：
 * ┌──────────────┬─────────────────────┬─────────────────────┐
 * │   资源       │  标志位=0 (fork)    │  标志位=1 (线程)    │
 * ├──────────────┼─────────────────────┼─────────────────────┤
 * │ mm (地址空间)│  dup_mm() 复制COW   │  mmget() 共享       │
 * │ files (fd表) │  dup_fd() 复制      │  引用计数+1 共享    │
 * │ fs (cwd等)   │  copy_fs_struct()   │  引用计数+1 共享    │
 * │ sighand      │  复制处理程序表     │  引用计数+1 共享    │
 * │ signal       │  分配新signal_struct│  直接共享           │
 * └──────────────┴─────────────────────┴─────────────────────┘
 */
static __latent_entropy struct task_struct *copy_process(
                    struct pid *pid,
                    int trace,
                    int node,
                    struct kernel_clone_args *args)
{
    int retval;
    struct task_struct *p;
    u64 clone_flags = args->flags;

    /*
     * ========== 阶段1: clone_flags合法性检查 ==========
     */
    
    /* 新挂载命名空间不能与父进程共享fs */
    if ((clone_flags & (CLONE_NEWNS|CLONE_FS)) == (CLONE_NEWNS|CLONE_FS))
        return ERR_PTR(-EINVAL);

    /* 
     * 线程依赖链: CLONE_THREAD → CLONE_SIGHAND → CLONE_VM
     * 线程必须共享信号处理，共享信号处理必须共享地址空间
     */
    if ((clone_flags & CLONE_THREAD) && !(clone_flags & CLONE_SIGHAND))
        return ERR_PTR(-EINVAL);
    if ((clone_flags & CLONE_SIGHAND) && !(clone_flags & CLONE_VM))
        return ERR_PTR(-EINVAL);

    /* 线程不能跨PID命名空间 */
    if (clone_flags & CLONE_THREAD) {
        if ((clone_flags & (CLONE_NEWUSER | CLONE_NEWPID)) ||
            (task_active_pid_ns(current) != nsp->pid_ns_for_children))
            return ERR_PTR(-EINVAL);
    }

    /*
     * ========== 阶段2: 分配task_struct ==========
     */
    
    /*
     * dup_task_struct(): 分配并初始化task_struct
     *   - 分配task_struct内存
     *   - 分配内核栈
     *   - 复制父进程的task_struct内容
     *   - 设置栈保护canary
     */
    p = dup_task_struct(current, node);
    if (!p)
        goto fork_out;

    /* 复制进程凭证(uid, gid, capabilities) */
    retval = copy_creds(p, clone_flags);
    if (retval < 0)
        goto bad_fork_free;

    /* 检查用户进程数限制(RLIMIT_NPROC) */
    if (is_ucounts_overlimit(..., RLIMIT_NPROC)) {
        if (!capable(CAP_SYS_RESOURCE))
            goto bad_fork_cleanup_count;
    }

    /* 检查系统线程总数限制(max_threads) */
    if (nr_threads >= max_threads)
        goto bad_fork_cleanup_count;

    /*
     * ========== 阶段3: 初始化基本字段 ==========
     */
    p->flags &= ~(PF_SUPERPRIV | PF_WQ_WORKER | PF_IDLE | PF_NO_SETAFFINITY);
    p->flags |= PF_FORKNOEXEC;

    INIT_LIST_HEAD(&p->children);
    INIT_LIST_HEAD(&p->sibling);

    /* 重置时间统计 */
    p->utime = p->stime = p->gtime = 0;
    p->start_time = ktime_get_ns();

    /*
     * ========== 阶段4: 复制/共享各种资源 ==========
     * 这是copy_process的核心部分
     */

    /* copy_semundo(): 复制System V信号量撤销信息 */
    retval = copy_semundo(clone_flags, p);
    if (retval) goto bad_fork_cleanup_security;

    /* copy_files(): 复制或共享文件描述符表 */
    retval = copy_files(clone_flags, p);
    if (retval) goto bad_fork_cleanup_semundo;

    /* copy_fs(): 复制或共享文件系统信息(cwd, root, umask) */
    retval = copy_fs(clone_flags, p);
    if (retval) goto bad_fork_cleanup_files;

    /* copy_sighand(): 复制或共享信号处理程序表 */
    retval = copy_sighand(clone_flags, p);
    if (retval) goto bad_fork_cleanup_fs;

    /* copy_signal(): 复制或共享signal_struct */
    retval = copy_signal(clone_flags, p);
    if (retval) goto bad_fork_cleanup_sighand;

    /* copy_mm(): 复制或共享地址空间 - 最重要！ */
    retval = copy_mm(clone_flags, p);
    if (retval) goto bad_fork_cleanup_signal;

    /* copy_namespaces(): 复制或创建新命名空间 */
    retval = copy_namespaces(clone_flags, p);
    if (retval) goto bad_fork_cleanup_mm;

    /* copy_io(): 复制I/O上下文 */
    retval = copy_io(clone_flags, p);
    if (retval) goto bad_fork_cleanup_namespaces;

    /*
     * copy_thread(): 复制线程上下文 - 架构相关
     * 设置子进程的:
     *   - 内核栈状态
     *   - 返回地址(ret_from_fork)
     *   - 返回值(AX=0，所以子进程fork返回0)
     *   - TLS
     */
    retval = copy_thread(clone_flags, args->stack, args->stack_size, p, args->tls);
    if (retval) goto bad_fork_cleanup_io;

    /*
     * ========== 阶段5: 分配PID ==========
     */
    pid = alloc_pid(p->nsproxy->pid_ns_for_children, ...);
    if (IS_ERR(pid))
        goto bad_fork_cleanup_thread;

    /*
     * ========== 阶段6: 设置PID和线程组关系 ==========
     */
    p->pid = pid_nr(pid);

    if (clone_flags & CLONE_THREAD) {
        /* 线程: 加入现有线程组 */
        p->group_leader = current->group_leader;
        p->tgid = current->tgid;
    } else {
        /* 进程: 成为新线程组的leader */
        p->group_leader = p;
        p->tgid = p->pid;
    }

    /*
     * ========== 阶段7: 设置父子关系 ==========
     */
    if (clone_flags & (CLONE_PARENT|CLONE_THREAD)) {
        p->real_parent = current->real_parent;
        if (clone_flags & CLONE_THREAD)
            p->exit_signal = -1;  // 线程退出不发信号
    } else {
        p->real_parent = current;
        p->exit_signal = args->exit_signal;  // 通常是SIGCHLD
    }

    /*
     * ========== 阶段8: 加入进程树和各种链表 ==========
     */
    write_lock_irq(&tasklist_lock);

    if (thread_group_leader(p)) {
        /* 进程: 加入父进程的children链表和全局进程链表 */
        list_add_tail(&p->sibling, &p->real_parent->children);
        list_add_tail_rcu(&p->tasks, &init_task.tasks);
        attach_pid(p, PIDTYPE_TGID);
        attach_pid(p, PIDTYPE_PGID);
        attach_pid(p, PIDTYPE_SID);
    } else {
        /* 线程: 加入线程组链表 */
        current->signal->nr_threads++;
        atomic_inc(&current->signal->live);
        list_add_tail_rcu(&p->thread_group, &p->group_leader->thread_group);
        list_add_tail_rcu(&p->thread_node, &p->signal->thread_head);
    }

    attach_pid(p, PIDTYPE_PID);
    nr_threads++;

    write_unlock_irq(&tasklist_lock);

    return p;

    /* 错误处理: 按相反顺序释放资源 */
bad_fork_cancel_cgroup:
    ...
bad_fork_free:
    ...
    return ERR_PTR(retval);
}
```

### 4.3 dup_task_struct - 分配任务描述符

```c
// kernel/fork.c:1044

/*
 * dup_task_struct() - 分配并初始化新的task_struct
 * 
 * 主要工作：
 *   1. 从slab缓存分配task_struct内存
 *   2. 分配内核栈(每个任务独立)
 *   3. 复制父进程的task_struct内容
 *   4. 初始化不能继承的字段
 * 
 * 返回：新分配的task_struct指针，或NULL
 */
static struct task_struct *dup_task_struct(struct task_struct *orig, int node)
{
    struct task_struct *tsk;
    unsigned long *stack;
    int err;

    /* 确定NUMA节点 */
    if (node == NUMA_NO_NODE)
        node = tsk_fork_get_node(orig);

    /*
     * 步骤1: 分配task_struct
     * alloc_task_struct_node(): 从task_struct slab缓存分配
     */
    tsk = alloc_task_struct_node(node);
    if (!tsk)
        return NULL;

    /*
     * 步骤2: 分配内核栈
     * 内核栈用于:
     *   - 系统调用时保存用户态寄存器
     *   - 中断/异常处理
     *   - 内核函数调用
     * 
     * 大小通常为: THREAD_SIZE (x86_64: 16KB = 4页)
     */
    stack = alloc_thread_stack_node(tsk, node);
    if (!stack)
        goto free_tsk;

    /*
     * 步骤3: 复制task_struct内容
     * arch_dup_task_struct(): 通常就是memcpy
     * 注意: 这是浅拷贝，指针字段指向相同的对象
     */
    err = arch_dup_task_struct(tsk, orig);

    /*
     * 步骤4: 重新设置栈指针
     * arch_dup_task_struct()复制了栈指针，必须更新为新栈
     */
    tsk->stack = stack;

    /*
     * 步骤5: 初始化thread_info
     * setup_thread_stack(): 复制thread_info或初始化
     */
    setup_thread_stack(tsk, orig);

    /*
     * 步骤6: 安全相关初始化
     */
#ifdef CONFIG_STACKPROTECTOR
    /* 生成新的栈保护canary，防止栈溢出攻击 */
    tsk->stack_canary = get_random_canary();
#endif

    /* 设置栈底魔数，用于检测栈溢出 */
    set_task_stack_end_magic(tsk);

    /* 清除需要重调度标志 */
    clear_tsk_need_resched(tsk);

    /*
     * 步骤7: 引用计数初始化
     */
    refcount_set(&tsk->rcu_users, 2);  // 用户空间 + 调度器
    refcount_set(&tsk->usage, 1);

    /* 清除不应继承的字段 */
    tsk->splice_pipe = NULL;
    tsk->task_frag.page = NULL;

    return tsk;

free_stack:
    free_thread_stack(tsk);
free_tsk:
    free_task_struct(tsk);
    return NULL;
}
```

### 4.4 copy_mm - 复制/共享地址空间

```c
// kernel/fork.c:1690

/*
 * copy_mm() - 复制或共享内存地址空间
 * 
 * 这是fork()和线程创建的核心区别所在！
 * 
 * CLONE_VM=0 (fork):
 *   调用dup_mm()创建地址空间副本
 *   使用COW(写时复制)优化，延迟实际内存复制
 * 
 * CLONE_VM=1 (线程/vfork):
 *   直接共享父进程的mm_struct
 *   只增加引用计数
 * 
 * 内存关系图：
 * 
 * fork()后:
 *   父进程mm_struct ──┐        子进程mm_struct
 *   ├─ pgd ──────────→│        ├─ pgd ─────────→ (独立页表)
 *   └─ VMAs           │        └─ VMAs           (独立VMA链表)
 *                     ↓                  ↓
 *               ┌─────────────────────────────┐
 *               │      物理页面(共享,只读)    │
 *               │      写时才复制             │
 *               └─────────────────────────────┘
 * 
 * 线程创建后:
 *   主线程mm_struct ←─────────→ 子线程mm_struct
 *        │                          │
 *        └──────── 同一个 ──────────┘
 *               mm_struct
 */
static int copy_mm(unsigned long clone_flags, struct task_struct *tsk)
{
    struct mm_struct *mm, *oldmm;

    /* 重置统计计数器 */
    tsk->min_flt = tsk->maj_flt = 0;  // 缺页统计
    tsk->nvcsw = tsk->nivcsw = 0;     // 上下文切换统计

    tsk->mm = NULL;
    tsk->active_mm = NULL;

    /*
     * 内核线程没有用户地址空间
     * 它们运行时会"借用"上一个用户进程的mm(称为active_mm)
     */
    oldmm = current->mm;
    if (!oldmm)
        return 0;

    vmacache_flush(tsk);  // 清空VMA查找缓存

    if (clone_flags & CLONE_VM) {
        /*
         * 共享模式: 用于创建线程或vfork
         * mmget(): 增加mm_struct引用计数
         * 
         * 结果: 父子任务的mm指针指向同一个mm_struct
         * 一方的内存修改立即对另一方可见
         */
        mmget(oldmm);
        mm = oldmm;
    } else {
        /*
         * 复制模式: 用于fork创建进程
         * dup_mm(): 创建独立的mm_struct副本
         * 
         * COW优化:
         *   1. 复制VMA结构(vm_area_struct链表)
         *   2. 复制页表，但不复制物理页面
         *   3. 将父子进程的页表项都标记为只读
         *   4. 首次写入时触发缺页，才真正复制页面
         */
        mm = dup_mm(tsk, current->mm);
        if (!mm)
            return -ENOMEM;
    }

    tsk->mm = mm;
    tsk->active_mm = mm;
    return 0;
}
```

### 4.5 copy_files - 复制/共享文件描述符表

```c
// kernel/fork.c:1784

/*
 * copy_files() - 复制或共享文件描述符表
 * 
 * files_struct结构：
 *   struct files_struct {
 *       atomic_t count;           // 引用计数
 *       struct fdtable *fdt;      // fd表指针
 *       struct file *fd_array[NR_OPEN_DEFAULT];  // 内嵌的小fd数组
 *   };
 * 
 * fd表中每个槽位指向一个struct file:
 *   fd[0] ──→ struct file (stdin)
 *   fd[1] ──→ struct file (stdout)
 *   fd[2] ──→ struct file (stderr)
 *   ...
 * 
 * CLONE_FILES=0 (fork):
 *   创建新的files_struct，复制fd数组
 *   每个struct file的引用计数+1
 *   父子进程可以独立close()
 * 
 * CLONE_FILES=1 (线程):
 *   共享同一个files_struct
 *   一个线程open()的文件对其他线程可见
 *   一个线程close()会影响其他线程
 */
static int copy_files(unsigned long clone_flags, struct task_struct *tsk)
{
    struct files_struct *oldf, *newf;
    int error = 0;

    oldf = current->files;
    if (!oldf)
        goto out;

    if (clone_flags & CLONE_FILES) {
        /*
         * 共享模式: 增加引用计数
         * 父子任务的files指针指向同一个files_struct
         */
        atomic_inc(&oldf->count);
        goto out;
    }

    /*
     * 复制模式: dup_fd()创建fd表副本
     * 每个打开的文件(struct file)引用计数+1
     */
    newf = dup_fd(oldf, NR_OPEN_MAX, &error);
    if (!newf)
        goto out;

    tsk->files = newf;
    error = 0;
out:
    return error;
}
```

### 4.6 copy_sighand - 复制/共享信号处理程序

```c
// kernel/fork.c:1854

/*
 * copy_sighand() - 复制或共享信号处理程序表
 * 
 * sighand_struct包含64个信号的处理程序设置:
 *   struct sighand_struct {
 *       refcount_t count;
 *       struct k_sigaction action[_NSIG];  // 64个信号处理程序
 *       spinlock_t siglock;
 *   };
 * 
 * k_sigaction包含:
 *   - sa_handler: 处理函数(SIG_DFL/SIG_IGN/用户函数)
 *   - sa_flags: SA_RESTART, SA_SIGINFO等标志
 *   - sa_mask: 处理期间阻塞的信号集
 * 
 * CLONE_SIGHAND=1 (线程必须):
 *   同一线程组的所有线程共享信号处理程序
 *   因为信号处理函数在共享的地址空间中
 */
static int copy_sighand(unsigned long clone_flags, struct task_struct *tsk)
{
    struct sighand_struct *sig;

    if (clone_flags & CLONE_SIGHAND) {
        /* 共享模式: 增加引用计数 */
        refcount_inc(&current->sighand->count);
        return 0;
    }

    /* 复制模式: 分配新的sighand_struct */
    sig = kmem_cache_alloc(sighand_cachep, GFP_KERNEL);
    RCU_INIT_POINTER(tsk->sighand, sig);
    if (!sig)
        return -ENOMEM;

    refcount_set(&sig->count, 1);

    /* 复制所有信号处理程序设置 */
    spin_lock_irq(&current->sighand->siglock);
    memcpy(sig->action, current->sighand->action, sizeof(sig->action));
    spin_unlock_irq(&current->sighand->siglock);

    /* CLONE_CLEAR_SIGHAND: 重置非SIG_IGN的处理程序 */
    if (clone_flags & CLONE_CLEAR_SIGHAND)
        flush_signal_handlers(tsk, 0);

    return 0;
}
```

### 4.7 copy_signal - 复制/共享signal_struct

```c
// kernel/fork.c:1925

/*
 * copy_signal() - 复制或共享signal_struct
 * 
 * signal_struct是线程组级别的结构，包含：
 *   - 线程组共享的待处理信号队列
 *   - 资源限制(RLIMIT_*)
 *   - CPU时间统计
 *   - 线程组中的线程计数
 *   - 子进程退出等待队列
 * 
 * CLONE_THREAD=1:
 *   线程共享signal_struct
 *   发送给进程的信号由线程组中任意线程处理
 * 
 * CLONE_THREAD=0:
 *   进程有独立的signal_struct
 *   各进程的信号队列相互独立
 */
static int copy_signal(unsigned long clone_flags, struct task_struct *tsk)
{
    struct signal_struct *sig;

    if (clone_flags & CLONE_THREAD)
        /* 线程: 复用父线程的signal_struct */
        return 0;

    /* 进程: 分配新的signal_struct */
    sig = kmem_cache_zalloc(signal_cachep, GFP_KERNEL);
    tsk->signal = sig;
    if (!sig)
        return -ENOMEM;

    /* 初始化线程组计数 */
    sig->nr_threads = 1;
    atomic_set(&sig->live, 1);
    refcount_set(&sig->sigcnt, 1);

    /* 
     * 初始化线程链表
     * thread_head <-> thread_node 形成双向循环链表
     */
    sig->thread_head = (struct list_head)LIST_HEAD_INIT(tsk->thread_node);
    tsk->thread_node = (struct list_head)LIST_HEAD_INIT(sig->thread_head);

    init_waitqueue_head(&sig->wait_chldexit);  // 子进程退出等待
    sig->curr_target = tsk;                    // 当前信号目标
    init_sigpending(&sig->shared_pending);     // 共享信号队列

    /* 继承父进程的资源限制 */
    memcpy(sig->rlim, current->signal->rlim, sizeof(sig->rlim));

    /* ... 其他初始化 ... */

    return 0;
}
```

### 4.8 copy_thread (x86_64) - 复制线程上下文

```c
// arch/x86/kernel/process.c:194

/*
 * copy_thread() - 复制线程上下文 (x86_64架构实现)
 * 
 * 这是fork能让子进程"返回"的关键！
 * 
 * 主要工作：
 *   1. 设置子进程的内核栈布局
 *   2. 设置返回地址为ret_from_fork
 *   3. 设置子进程的返回值为0 (通过设置AX寄存器)
 *   4. 复制用户态寄存器状态
 *   5. 设置TLS
 * 
 * 内核栈布局 (子进程首次被调度时):
 * 
 *   高地址
 *   ┌─────────────────────┐
 *   │     pt_regs         │ ← 用户态寄存器(包括AX=0)
 *   │  (用户态上下文)     │
 *   ├─────────────────────┤
 *   │  fork_frame         │ ← 包含ret_addr = ret_from_fork
 *   │  (内核返回框架)     │
 *   ├─────────────────────┤
 *   │                     │
 *   │   (可用栈空间)      │
 *   │                     │
 *   └─────────────────────┘
 *   低地址
 * 
 * 子进程首次调度执行流程:
 *   schedule() → context_switch() → switch_to()
 *       │
 *       ↓
 *   从子进程内核栈恢复寄存器
 *       │
 *       ↓
 *   ret指令 → 跳转到ret_from_fork
 *       │
 *       ↓
 *   ret_from_fork恢复pt_regs并返回用户态
 *       │
 *       ↓
 *   用户态看到fork()返回0
 */
int copy_thread(unsigned long clone_flags, unsigned long sp, unsigned long arg,
        struct task_struct *p, unsigned long tls)
{
    struct inactive_task_frame *frame;
    struct fork_frame *fork_frame;
    struct pt_regs *childregs;

    /* 获取子进程内核栈顶部的pt_regs区域 */
    childregs = task_pt_regs(p);
    fork_frame = container_of(childregs, struct fork_frame, regs);
    frame = &fork_frame->frame;

    /* 设置栈帧基址指针 */
    frame->bp = encode_frame_pointer(childregs);
    
    /*
     * 关键! 设置返回地址为ret_from_fork
     * 子进程首次被调度时，会从这里"返回"
     */
    frame->ret_addr = (unsigned long) ret_from_fork;
    
    /* 设置子进程的栈指针 */
    p->thread.sp = (unsigned long) fork_frame;
    
    /* 清除调试断点 */
    memset(p->thread.ptrace_bps, 0, sizeof(p->thread.ptrace_bps));

#ifdef CONFIG_X86_64
    /* 保存段寄存器(FS/GS用于TLS) */
    current_save_fsgs();
    p->thread.fsindex = current->thread.fsindex;
    p->thread.fsbase = current->thread.fsbase;
    p->thread.gsindex = current->thread.gsindex;
    p->thread.gsbase = current->thread.gsbase;
#endif

    /* 内核线程特殊处理 */
    if (unlikely(p->flags & PF_KTHREAD)) {
        memset(childregs, 0, sizeof(struct pt_regs));
        kthread_frame_init(frame, sp, arg);
        return 0;
    }

    /*
     * 复制父进程的用户态寄存器到子进程
     * 这样子进程恢复执行时有正确的上下文
     */
    *childregs = *current_pt_regs();
    
    /*
     * 关键! 设置子进程的返回值为0
     * 这就是为什么 if (fork() == 0) 能区分父子进程
     */
    childregs->ax = 0;

    /* 如果提供了用户栈指针(线程创建)，更新SP */
    if (sp)
        childregs->sp = sp;

    /* 设置TLS */
    if (clone_flags & CLONE_SETTLS)
        set_new_tls(p, tls);

    return 0;
}
```

### 4.9 wake_up_new_task - 唤醒新任务

```c
// kernel/sched/core.c:4493

/*
 * wake_up_new_task() - 唤醒新创建的进程/线程
 * 
 * 这是新任务"开始存在"的时刻！
 * 
 * 主要工作：
 *   1. 设置任务状态为TASK_RUNNING
 *   2. 选择最合适的CPU运行队列
 *   3. 将任务加入运行队列
 *   4. 检查是否需要抢占当前任务
 * 
 * 调用此函数后，新任务随时可能被调度执行
 */
void wake_up_new_task(struct task_struct *p)
{
    struct rq_flags rf;
    struct rq *rq;

    raw_spin_lock_irqsave(&p->pi_lock, rf.flags);
    
    /*
     * 设置任务状态为TASK_RUNNING
     * 这是任务能被调度的前提
     */
    WRITE_ONCE(p->__state, TASK_RUNNING);

#ifdef CONFIG_SMP
    /*
     * SMP负载均衡：选择最合适的CPU
     * 考虑因素:
     *   - 父进程所在CPU(缓存亲和性)
     *   - 各CPU负载情况
     *   - NUMA拓扑
     */
    p->recent_used_cpu = task_cpu(p);
    __set_task_cpu(p, select_task_rq(p, task_cpu(p), WF_FORK));
#endif

    rq = __task_rq_lock(p, &rf);
    update_rq_clock(rq);
    
    /* 初始化调度实体的平均利用率 */
    post_init_entity_util_avg(p);

    /*
     * activate_task(): 将任务加入运行队列
     * 对于CFS调度器，这会将任务插入红黑树
     */
    activate_task(rq, p, ENQUEUE_NOCLOCK);
    
    trace_sched_wakeup_new(p);
    
    /*
     * check_preempt_curr(): 检查新任务是否应该抢占当前任务
     * 如果新任务优先级更高，设置need_resched标志
     * 当前任务会在下一个调度点让出CPU
     */
    check_preempt_curr(rq, p, WF_FORK);

#ifdef CONFIG_SMP
    if (p->sched_class->task_woken)
        p->sched_class->task_woken(rq, p);
#endif

    task_rq_unlock(rq, p, &rf);
}
```

```
    │              │                           │           │
    │              │                     切换到子进程      │
    │              │                           │           │
    │              │                     ret_from_fork     │
    │              │                           │           │
    │              │                     返回用户态        │
    │              │                           │           │
    │              │    子进程看到fork()返回0 ←┘           │
    │              │                                       │
```

## 七、总结

Linux进程/线程创建的核心设计理念：

1. **统一抽象**: 进程和线程都是`task_struct`，区别仅在于资源共享程度

2. **写时复制(COW)**: `fork()`不立即复制内存，而是共享物理页并延迟到写入时才复制

3. **灵活的`clone()`**: 通过`clone_flags`位图，可以精确控制哪些资源共享、哪些复制

4. **线程组抽象**: 同一进程的多个线程通过共享`signal_struct`和`mm_struct`，对外表现为单一进程

5. **命名空间隔离**: 容器技术的基础，通过`CLONE_NEW*`标志创建隔离的资源视图

6. **零拷贝启动**: 新任务创建后立即可被调度，无需等待资源复制完成
